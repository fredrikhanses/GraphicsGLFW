#version 330 core
uniform vec2 u_MidPoint;
uniform float u_Time;
uniform float u_Scale = 1.0f;
uniform vec2 u_Offset;
in vec2 a_Position;
void main()
{
    vec2 correction = (1.0f - u_Scale) * (u_MidPoint - a_Position);
    correction = a_Position + correction + u_Offset;
    gl_Position = vec4(correction, 0.0f, 1.0f);
}
